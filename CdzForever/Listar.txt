    /remini autoluta\ $+ $nick $+ .ini $nick Elevou

    /remini op.ini $nick 
    /remini admin.ini $nick
    /remini listando.ini $nick
    /close -c $nick 
  }
  /set %opennum 1
  while (%opennum <= $chat(0)) {
    /set %nick $chat(%opennum)
    if ( %nick != $nick) { 
      /msg = $+ %nick 2( $nick 14Saiu do 2Chat )
    }
    inc %opennum
  }
}
