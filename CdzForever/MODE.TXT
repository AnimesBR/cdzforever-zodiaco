off *:chat:.Mode +m: {
  if ( $readini(cadastros\ $+ $nick $+ .ini,$nick,nick) == $null ) {
  /msg =$nick 2(4 ERRO:14 Você não está Cadastrado 2) | halt }
    if ( $readini(cadastros\ $+ $nick $+ .ini,$nick,id) == 0 ) {
      /msg =$nick 2(14 Você não esta Identificado2,14 Digite2: .14Login 2<14senha2> )
      .halt
    }
    if ( $readini(cadastros\ $+ $nick $+ .ini,$nick,id) == $null ) {
      /msg =$nick 2(14 Você não esta Identificado2,14 Digite2: .14Login 2<14senha2> )
      .halt
    }
    if ( $readini(cadastros\ $+ $nick $+ .ini,$nick,id) == 1 ) {
    if !$2 { /msg =$nick 2(14 Comando Inválido 2) | halt }
    /writeini modes.ini $nick mode sim
    /msg =$nick 2(14 Você ativou modo 2+m 2)
  }
}
